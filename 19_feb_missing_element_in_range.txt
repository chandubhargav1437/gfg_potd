class Solution {
  public:
    vector<int> missinRange(vector<int>& arr, int low, int high) {
        vector<int>ans;
        map<int,int>mp;
        int loww=low,highh=high;
        for(auto x:arr){
            if(x>=low&&x<=high){
                mp[x]++;
            }
        }
        for(int i=low;i<=high;i++){
            if(mp[i]==0){
                ans.push_back(i);
            }
        }
        return ans;
    }
};

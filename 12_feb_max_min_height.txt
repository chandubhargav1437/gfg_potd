#include <vector>
#include <algorithm>
#include <iostream>

using namespace std;

class Solution {
public:
    // Helper function to check if a specific minimum height is possible
    bool isPossible(vector<int> &arr, int n, int k, int w, int targetHeight) {
        // 'add' stores how much water we started adding at specific indices.
        // We use a long long because the water amount can exceed integer limits.
        vector<long long> add(n, 0); 
        
        long long current_added_water = 0; // Tracks active water effect from sliding windows
        long long days_needed = 0;
        
        for (int i = 0; i < n; i++) {
            // 1. Remove the effect of the window that just went out of scope.
            // A window starting at (i - w) stops affecting index i.
            if (i >= w) {
                current_added_water -= add[i - w];
            }

            // 2. Calculate current height of flower i
            // Base height + water from active windows
            long long current_height = arr[i] + current_added_water;

            // 3. If current flower is still too short, we must water it
            if (current_height < targetHeight) {
                long long needed = targetHeight - current_height;
                
                days_needed += needed;
                
                // If we exceed the total allowed days k, this target is impossible
                if (days_needed > k) return false;

                // We start a new watering window here at i
                add[i] = needed;
                current_added_water += needed;
            }
        }
        return true;
    }

    int maxMinHeight(vector<int> &arr, int k, int w) {
        int n = arr.size();
        
        // Initialize Binary Search bounds
        // Low: The minimum existing height (we can definitely achieve this)
        // High: The maximum existing height + k (theoretical max if we dumped all water on max)
        int low = *min_element(arr.begin(), arr.end());
        int high = *max_element(arr.begin(), arr.end()) + k;
        
        int ans = low;

        while (low <= high) {
            int mid = low + (high - low) / 2;
            
            if (isPossible(arr, n, k, w, mid)) {
                // If mid is possible, try for a higher minimum
                ans = mid;
                low = mid + 1;
            } else {
                // If mid is not possible, we need to lower our target
                high = mid - 1;
            }
        }
        
        return ans;
    }
};

#include <vector>
#include <cmath>
#include <climits>

using namespace std;

class Solution {
  public:
    vector<int> findClosestPair(vector<int> &arr1, vector<int> &arr2, int x) {
        int n = arr1.size();
        int m = arr2.size();
        
        int l = 0;           // Pointer for arr1 (starts at smallest)
        int r = m - 1;       // Pointer for arr2 (starts at largest)
        
        int min_diff = INT_MAX;
        int res_l = -1, res_r = -1;
        
        while (l < n && r >= 0) {
            int current_sum = arr1[l] + arr2[r];
            int current_diff = abs(current_sum - x);
            
            // If we found a strictly closer pair, update our trackers
            if (current_diff < min_diff) {
                min_diff = current_diff;
                res_l = arr1[l];
                res_r = arr2[r];
            }
            
            // Adjust pointers based on how the sum compares to x
            if (current_sum == x) {
                // Difference is exactly 0, can't get closer than this
                return {arr1[l], arr2[r]};
            } else if (current_sum < x) {
                l++; // Need a bigger sum
            } else {
                r--; // Need a smaller sum
            }
        }
        
        return {res_l, res_r};
    }
};

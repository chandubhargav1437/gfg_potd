class Solution {
  public:
    int minCandy(vector<int> &arr) {
        // code here
        int n = arr.size();
        
        // Edge case: if there are no children, 0 candies needed.
        if (n == 0) return 0;

        // Step 1: Give every child 1 candy initially.
        // This satisfies the condition "Each child must have at least one candy."
        vector<int> candies(n, 1);

        // Step 2: Left-to-Right Pass
        // Check if the current child has a higher rating than the LEFT neighbor.
        for (int i = 1; i < n; i++) {
            if (arr[i] > arr[i - 1]) {
                candies[i] = candies[i - 1] + 1;
            }
        }

        // Step 3: Right-to-Left Pass
        // Check if the current child has a higher rating than the RIGHT neighbor.
        for (int i = n - 2; i >= 0; i--) {
            if (arr[i] > arr[i + 1]) {
                // We use max() to keep the higher value.
                // If we simply assigned (candies[i+1] + 1), we might break 
                // the condition satisfied in the Left-to-Right pass.
                candies[i] = max(candies[i], candies[i + 1] + 1);
            }
        }

        // Step 4: Calculate total candies
        // accumulate sums from beginning to end, starting with 0.
        int totalCandies = accumulate(candies.begin(), candies.end(), 0);

        return totalCandies;
    }
};

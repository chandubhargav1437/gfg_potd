#include <vector>
#include <algorithm>

class Solution {
public:
    int hIndex(std::vector<int>& citations) {
        // Step 1: Sort citations in descending order
        std::sort(citations.begin(), citations.end(), std::greater<int>());
        
        int h = 0;
        // Step 2: Iterate and compare citation count with the paper rank
        while (h < citations.size() && citations[h] > h) {
            h++;
        }
        
        return h;
    }
};

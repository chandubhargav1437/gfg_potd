#include <vector>
#include <unordered_map>
#include <algorithm>

using namespace std;

class Solution {
public:
    int equalSumSpan(vector<int> &a1, vector<int> &a2) {
        // Map to store the first occurrence of a prefix sum
        unordered_map<int, int> preSumMap;
        
        int max_len = 0;
        int curr_sum = 0;
        
        // Initialize for cases where the valid span starts from index 0
        preSumMap[0] = -1;
        
        for (int i = 0; i < a1.size(); i++) {
            // Calculate the running sum of differences
            curr_sum += (a1[i] - a2[i]);
            
            // If we've seen this running sum before, the subarray between 
            // the previous occurrence and the current index sums to 0.
            if (preSumMap.find(curr_sum) != preSumMap.end()) {
                max_len = max(max_len, i - preSumMap[curr_sum]);
            } else {
                // Only store the FIRST time we see a sum to maximize the length
                preSumMap[curr_sum] = i;
            }
        }
        
        return max_len;
    }
};

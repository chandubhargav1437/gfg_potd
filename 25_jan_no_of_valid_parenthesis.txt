class Solution {
  public:
    int findWays(int n) {
        // code here
        // A valid parentheses expression must have an equal number of '(' and ')'
        // so the length 'n' must be even.
        if (n % 2 != 0) {
            return 0;
        }

        // We need to find the k-th Catalan number, where k is the number of pairs.
        int k = n / 2;

        // Use long long to prevent integer overflow during calculation
        long long res = 1;

        // Calculate Binomial Coefficient C(2k, k) iteratively
        // Formula used: C_k = (1 / (k + 1)) * (2k choose k)
        for (int i = 0; i < k; i++) {
            res = res * (2 * k - i);
            res = res / (i + 1);
        }

        // Apply the 1/(k+1) factor
        return (int)(res / (k + 1));
        
    }
};

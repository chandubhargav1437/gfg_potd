#include <vector>
#include <algorithm> // for swap

using namespace std;

class Solution {
  private:
    // Helper function to perform backtracking
    void solve(vector<int>& arr, int index, vector<vector<int>>& ans) {
        // Base Case: If the current index reaches the end of the array,
        // we have formed a valid permutation.
        if (index == arr.size()) {
            ans.push_back(arr);
            return;
        }

        // Iterate from the current index to the end of the array
        for (int i = index; i < arr.size(); i++) {
            // Swap the element at the current index with the loop index
            swap(arr[index], arr[i]);
            
            // Recursively call for the next position
            solve(arr, index + 1, ans);
            
            // Backtracking: Swap the elements back to restore the original 
            // array state for the next iteration of the loop
            swap(arr[index], arr[i]);
        }
    }

  public:
    vector<vector<int>> permuteDist(vector<int>& arr) {
        vector<vector<int>> ans;
        // Start recursion from index 0
        solve(arr, 0, ans);
        return ans;
    }
};

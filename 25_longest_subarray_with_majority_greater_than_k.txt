#include <vector>
#include <unordered_map>
#include <algorithm>

using namespace std;

class Solution {
public:
    int longestSubarray(vector<int>& arr, int k) {
        // Map to store the earliest index of each prefix sum
        unordered_map<int, int> first_occ;
        int sum = 0;
        int max_len = 0;

        for (int i = 0; i < arr.size(); i++) {
            // Transform dynamically: +1 if > k, -1 otherwise
            sum += (arr[i] > k) ? 1 : -1;

            if (sum > 0) {
                // The whole prefix from start is valid
                max_len = i + 1;
            } else {
                // Look for the earliest occurrence of sum - 1
                if (first_occ.find(sum - 1) != first_occ.end()) {
                    max_len = max(max_len, i - first_occ[sum - 1]);
                }
            }

            // Only record the first time we see a particular sum 
            // to maximize the length of future subarrays
            if (first_occ.find(sum) == first_occ.end()) {
                first_occ[sum] = i;
            }
        }

        return max_len;
    }
};

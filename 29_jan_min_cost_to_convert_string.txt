#include <iostream>
#include <string>
#include <vector>
#include <queue>

using namespace std;

class Solution {
  public:
    string firstNonRepeating(string &s) {
        // Frequency array for lowercase alphabets 'a'-'z'
        vector<int> count(26, 0);
        // Queue to maintain the order of characters
        queue<char> q;
        string result = "";
        
        for (char c : s) {
            // 1. Update frequency
            count[c - 'a']++;
            
            // 2. Push current character to queue
            q.push(c);
            
            // 3. Remove characters from the front that have repeated
            // We check the frequency of the character currently at the front
            while (!q.empty() && count[q.front() - 'a'] > 1) {
                q.pop();
            }
            
            // 4. Append result
            if (q.empty()) {
                result += '#';
            } else {
                result += q.front();
            }
        }
        
        return result;
    }
};
